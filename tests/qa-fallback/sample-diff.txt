+import { db } from '../db';
+
+const API_KEY = "sk-1234567890abcdef";  // hardcoded secret (intentional flaw)
+
+export async function login(username: string, password: string) {
+  // SQL injection vulnerability (intentional flaw)
+  const query = `SELECT * FROM users WHERE name = '${username}' AND pw = '${password}'`;
+  const user = await db.raw(query);
+
+  if (!user) {
+    return null;
+  }
+
+  return {
+    token: API_KEY + user.id,
+    user,
+  };
